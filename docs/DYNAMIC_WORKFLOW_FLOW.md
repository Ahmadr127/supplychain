â””â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”˜

// Saat request dibuat atau harga diupdate:

function evaluateAllSteps(ApprovalRequest $request) {
    $totalValue = $request->items->sum('total_price');
    
    foreach ($request->itemSteps as $step) {
        if (!$step->is_conditional) {
            continue; // Non-conditional steps always active
        }
        
        $shouldBeActive = evaluateCondition($step, $request, $totalValue);
        
        if ($shouldBeActive && $step->status === 'skipped') {
            // Reactivate step
            $step->update([
                'status' => 'pending',
                'skip_reason' => null
            ]);
        } elseif (!$shouldBeActive && $step->status === 'pending') {
            // Skip step
            $step->update([
                'status' => 'skipped',
                'skip_reason' => 'Kondisi tidak terpenuhi: ' . $step->condition_type
            ]);
        }
    }
}

function evaluateCondition($step, $request, $totalValue) {
    switch ($step->condition_type) {
        case 'total_price_min':
            return $totalValue >= $step->condition_value;
            
        case 'total_price_max':
            return $totalValue < $step->condition_value;
            
        case 'total_price_between':
            $values = json_decode($step->condition_value);
            return $totalValue >= $values->min && $totalValue < $values->max;
            
        case 'classification':
            return $request->classification_id == $step->condition_value;
            
        case 'department':
            return $request->requester->primaryDepartment->id == $step->condition_value;
            
        case 'item_type':
            return $request->item_type_id == $step->condition_value;
            
        default:
            return true; // Default: active
    }
}

function processApproval(ApprovalItemStep $currentStep, User $approver) {
    DB::transaction(function() use ($currentStep, $approver) {
        
        // 1. Mark current step as approved
        $currentStep->update([
            'status' => 'approved',
            'approved_by' => $approver->id,
            'approved_at' => now()
        ]);
        
        // 2. Find next active step (skip any skipped steps)
        $nextStep = ApprovalItemStep::where('approval_request_id', $currentStep->approval_request_id)
            ->where('master_item_id', $currentStep->master_item_id)
            ->where('step_number', '>', $currentStep->step_number)
            ->whereIn('status', ['pending']) // Only pending steps
            ->orderBy('step_number')
            ->first();
        
        if (!$nextStep) {
            // Check if there are skipped steps that should be evaluated
            reevaluateSkippedSteps($currentStep->approval_request_id, $currentStep->master_item_id);
            
            $nextStep = ApprovalItemStep::where('approval_request_id', $currentStep->approval_request_id)
                ->where('master_item_id', $currentStep->master_item_id)
                ->where('step_number', '>', $currentStep->step_number)
                ->where('status', 'pending')
                ->orderBy('step_number')
                ->first();
        }
        
        if (!$nextStep) {
            // All steps completed - mark item as approved
            markItemAsApproved($currentStep->master_item_id, $currentStep->approval_request_id);
        }
        
        // 3. Aggregate request status
        aggregateRequestStatus($currentStep->approval_request_id);
    });
}
